create table public.admissions (

  program_type character varying(100) null,
  degree_level character varying(100) null,
  field_of_study character varying(255) null,
  duration character varying(100) null,
  tuition_fee numeric(12, 2) null,
  currency character varying(3) null,
  application_fee numeric(10, 2) null,
  deadline timestamp with time zone null,
  start_date date null,
  location character varying(255) null,
  delivery_mode character varying(50) null,
  requirements jsonb null,
  verification_status public.verification_status not null default 'draft'::verification_status,
  verified_at timestamp with time zone null,
  verified_by uuid null,
  rejection_reason text null,
  review_reason text null,
  created_by uuid null,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone not null default now(),
  is_active boolean not null default true,
  admin_notes text null,
  verification_comments text null,
  needs_reverification boolean null default false,
  updated_by uuid null,
  constraint admissions_pkey primary key (id),
  constraint admissions_updated_by_fkey foreign KEY (updated_by) references users (id),
  constraint check_review_reason check (
    (
      (
        (
          verification_status = 'rejected'::verification_status
        )
        and (review_reason is not null)
      )
      or (
        verification_status <> 'rejected'::verification_status
      )
    )
  ),
  constraint check_rejection_reason check (
    (
      (
        (
          verification_status = 'rejected'::verification_status
        )
        and (rejection_reason is not null)
      )
      or (
        verification_status <> 'rejected'::verification_status
      )
    )
  ),
  constraint check_verified_fields check (
    (
      (
        (
          verification_status = 'verified'::verification_status
        )
        and (verified_at is not null)
      )
      or (
        verification_status <> 'verified'::verification_status
      )
    )
  )
) TABLESPACE pg_default;
